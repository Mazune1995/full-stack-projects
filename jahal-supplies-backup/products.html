// Fetch dynamic products from backend with images fetch("http://127.0.0.1:8000/api/products/") .then(response => response.json()) .then(products => { const container = document.getElementById("products"); container.innerHTML = ""; if (products.length === 0) { container.innerHTML = "

No products available yet.

"; return; } products.forEach(product => { const item = document.createElement("div"); item.className = "product-card"; item.innerHTML = `

${product.name}

Price: $${product.price}

${product.description}

`; container.appendChild(item); }); }) .catch(error => console.error("Error fetching products:", error));
